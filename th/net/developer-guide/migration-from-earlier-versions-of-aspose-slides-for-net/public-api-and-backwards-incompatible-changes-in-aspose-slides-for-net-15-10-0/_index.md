## **การเปลี่ยนแปลง Public API**
#### **เพิ่ม VideoPlayerHtmlController ใหม่เพื่อสนับสนุนการส่งออกไฟล์สื่อเป็น HTML**
คลาสสาธารณะใหม่ VideoPlayerHtmlController ได้ถูกเพิ่มเข้าไปในเนมสเปซ Aspose.Slides.Export ผู้ใช้สามารถใช้อินสแตนซ์ของคลาสนี้เพื่อส่งออกไฟล์วิดีโอและเสียงเป็น HTML ได้
คอนสตรัคเตอร์ของ VideoPlayerHtmlController รับพารามิเตอร์ต่อไปนี้:

path: เส้นทางที่ไฟล์วิดีโอและเสียงจะถูกสร้าง  
fileName: ชื่อของไฟล์ HTML  
baseUri: URI พื้นฐานที่ใช้สำหรับสร้างลิงก์  
ตัวอย่างการใช้งาน:

``` csharp

 using (Presentation pres = new Presentation("example.pptx"))

{

    const string path = "path";

    const string fileName = "video.html";

    const string baseUri = "http://www.example.com/";

    VideoPlayerHtmlController controller = new VideoPlayerHtmlController(path: path, fileName: fileName, baseUri: baseUri);

    HtmlOptions htmlOptions = new HtmlOptions(controller);

    SVGOptions svgOptions = new SVGOptions(controller);

    htmlOptions.HtmlFormatter = HtmlFormatter.CreateCustomFormatter(controller);

    htmlOptions.SlideImageFormat = SlideImageFormat.Svg(svgOptions);

    pres.Save(Path.Combine(path, fileName), SaveFormat.Html, htmlOptions);

}

``` 
#### **เพิ่ม API การเคลื่อนไหวของซีรีส์แผนภูมิ**
เมธอดใหม่ 2 เมธอดได้ถูกเพิ่มเข้าไปในอินเทอร์เฟซ Aspose.Slides.Animation.ISequence interface.

``` csharp

 IEffect AddEffect(IChart chart, EffectChartMajorGroupingType type, int index, EffectType effectType, EffectSubtype subtype, EffectTriggerType triggerType);

IEffect AddEffect(IChart chart, EffectChartMinorGroupingType type, int seriesIndex, int categoriesIndex, EffectType effectType, EffectSubtype subtype, EffectTriggerType triggerType);

``` 

เมธอดเหล่านี้มีจุดประสงค์เพื่อสนับสนุนการเคลื่อนไหวขององค์ประกอบแผนภูมิ:
- ตามซีรีส์
- ตามประเภท
- ตามองค์ประกอบของซีรีส์
- ตามองค์ประกอบของประเภท

ได้แนะนำสอง enum ใหม่คือ EffectChartMajorGroupingType และ EffectChartMinorGroupingType ที่เกี่ยวข้องกับการเคลื่อนไหวขององค์ประกอบแผนภูมิ

เพื่อเพิ่มการเคลื่อนไหวของซีรีส์ให้กับแผนภูมิ สามารถใช้โค้ดต่อไปนี้:

``` csharp

 using (Presentation pres = new Presentation(inFileName))

{

    var slide = pres.Slides[0] as Slide;

    var shapes = slide.Shapes as ShapeCollection;

    var chart = shapes[0] as IChart;

    slide.Timeline.MainSequence.AddEffect(chart, EffectType.Fade, EffectSubtype.None,

        EffectTriggerType.AfterPrevious);

    ((Sequence)slide.Timeline.MainSequence).AddEffect(chart,

        EffectChartMajorGroupingType.BySeries, 0,

        EffectType.Appear, EffectSubtype.None, EffectTriggerType.AfterPrevious);

    ((Sequence)slide.Timeline.MainSequence).AddEffect(chart,

        EffectChartMajorGroupingType.BySeries, 1,

        EffectType.Appear, EffectSubtype.None, EffectTriggerType.AfterPrevious);

    ((Sequence)slide.Timeline.MainSequence).AddEffect(chart,

        EffectChartMajorGroupingType.BySeries, 2,

        EffectType.Appear, EffectSubtype.None, EffectTriggerType.AfterPrevious);

    ((Sequence)slide.Timeline.MainSequence).AddEffect(chart,

        EffectChartMajorGroupingType.BySeries, 3,

        EffectType.Appear, EffectSubtype.None, EffectTriggerType.AfterPrevious);

    pres.Save(outFileName, SaveFormat.Pptx);

}

```
