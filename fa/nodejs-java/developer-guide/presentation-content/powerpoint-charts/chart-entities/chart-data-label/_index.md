## **Щ…ЩӮШҜЩ…ЩҮ**

ШЁШұЪҶШіШЁвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮ ШҜШұ ЩҶЩ…ЩҲШҜШ§Шұ Ш¬ШІШҰЫҢШ§ШӘЫҢ ШҜШұШЁШ§ШұЩҮЩ” ШіШұЫҢвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ ЫҢШ§ ЩҶЩӮШ§Ш· ШҜШ§ШҜЩҮЩ” ЩҒШұШҜЫҢ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮЩҶШҜ. Ш§ЫҢЩҶ ШЁШұЪҶШіШЁвҖҢЩҮШ§ ШЁЩҮ Ш®ЩҲШ§ЩҶЩҶШҜЪҜШ§ЩҶ Ш§Щ…Ъ©Ш§ЩҶ Щ…ЫҢвҖҢШҜЩҮЩҶШҜ ШӘШ§ ШіШұЫҢвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮ ШұШ§ ШЁЩҮвҖҢШіШұШ№ШӘ ШҙЩҶШ§ШіШ§ЫҢЫҢ Ъ©ЩҶЩҶШҜ ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ШҜШұЪ© ЩҶЩ…ЩҲШҜШ§ШұЩҮШ§ ШұШ§ ШўШіШ§ЩҶвҖҢШӘШұ Щ…ЫҢвҖҢШіШ§ШІЩҶШҜ.

## **ШӘЩҶШёЫҢЩ… ШҜЩӮШӘ ШҜШ§ШҜЩҮвҖҢЩҮШ§ ШҜШұ ШЁШұЪҶШіШЁвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ**

Ш§ЫҢЩҶ Ъ©ШҜ Ш¬Ш§ЩҲШ§Ш§ШіЪ©ШұЫҢЩҫШӘ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ ЪҶЪҜЩҲЩҶЩҮ ШҜЩӮШӘ ШҜШ§ШҜЩҮвҖҢЩҮШ§ ШұШ§ ШҜШұ ЫҢЪ© ШЁШұЪҶШіШЁ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЫҢШҜ:
```javascript
var pres = new aspose.slides.Presentation();
try {
    var chart = pres.getSlides().get_Item(0).getShapes().addChart(aspose.slides.ChartType.Line, 50, 50, 450, 300);
    chart.setDataTable(true);
    chart.getChartData().getSeries().get_Item(0).setNumberFormatOfValues("#,##0.00");
    pres.save("output.pptx", aspose.slides.SaveFormat.Pptx);
} finally {
    if (pres != null) {
        pres.dispose();
    }
}
```

## **ЩҶЩ…Ш§ЫҢШҙ ШҜШұШөШҜ ШЁЩҮвҖҢШөЩҲШұШӘ ШЁШұЪҶШіШЁвҖҢЩҮШ§**

Aspose.Slides ШЁШұШ§ЫҢ Node.js Ш§ШІ Ш·ШұЫҢЩӮ Java ШЁЩҮ ШҙЩ…Ш§ Ш§Щ…Ъ©Ш§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ ШЁШұЪҶШіШЁвҖҢЩҮШ§ЫҢ ШҜШұШөШҜЫҢ ШұШ§ ШұЩҲЫҢ ЩҶЩ…ЩҲШҜШ§ШұЩҮШ§ЫҢ ЩҶЩ…Ш§ЫҢШҙ ШҜШ§ШҜЩҮ ШҙШҜЩҮ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЫҢШҜ. Ш§ЫҢЩҶ Ъ©ШҜ Ш¬Ш§ЩҲШ§Ш§ШіЪ©ШұЫҢЩҫШӘ Ш№Щ…Щ„Ъ©ШұШҜ ШұШ§ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ:
```javascript
// ЫҢЪ© ЩҶЩ…ЩҲЩҶЩҮ Ш§ШІ Ъ©Щ„Ш§Ші Presentation Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ
var pres = new aspose.slides.Presentation();
try {
    // Ш§ЩҲЩ„ЫҢЩҶ Ш§ШіЩ„Ш§ЫҢШҜ ШұШ§ ШҜШұЫҢШ§ЩҒШӘ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    var slide = pres.getSlides().get_Item(0);
    var chart = slide.getShapes().addChart(aspose.slides.ChartType.StackedColumn, 20, 20, 400, 400);
    var series;
    var total_for_Cat = new double[chart.getChartData().getCategories().size()];
    for (var k = 0; k < chart.getChartData().getCategories().size(); k++) {
        var cat = chart.getChartData().getCategories().get_Item(k);
        for (var i = 0; i < chart.getChartData().getSeries().size(); i++) {
            total_for_Cat[k] = total_for_Cat[k] + chart.getChartData().getSeries().get_Item(i).getDataPoints().get_Item(k).getValue().getData();
        }
    }
    var dataPontPercent = 0.0;
    for (var x = 0; x < chart.getChartData().getSeries().size(); x++) {
        series = chart.getChartData().getSeries().get_Item(x);
        series.getLabels().getDefaultDataLabelFormat().setShowLegendKey(false);
        for (var j = 0; j < series.getDataPoints().size(); j++) {
            var lbl = series.getDataPoints().get_Item(j).getLabel();
            dataPontPercent = (series.getDataPoints().get_Item(j).getValue().getData() / total_for_Cat[j]) * 100;
            var port = new aspose.slides.Portion();
            port.setText(java.callStaticMethodSync("java.lang.String", "format", "{0:F2} %.2f", dataPontPercent));
            port.getPortionFormat().setFontHeight(8.0);
            lbl.getTextFrameForOverriding().setText("");
            var para = lbl.getTextFrameForOverriding().getParagraphs().get_Item(0);
            para.getPortions().add(port);
            lbl.getDataLabelFormat().setShowSeriesName(false);
            lbl.getDataLabelFormat().setShowPercentage(false);
            lbl.getDataLabelFormat().setShowLegendKey(false);
            lbl.getDataLabelFormat().setShowCategoryName(false);
            lbl.getDataLabelFormat().setShowBubbleSize(false);
        }
    }
    // Ш§ШұШ§ШҰЩҮ ШӯШ§ЩҲЫҢ ЩҶЩ…ЩҲШҜШ§Шұ ШұШ§ Ш°Ш®ЫҢШұЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    pres.save("output.pptx", aspose.slides.SaveFormat.Pptx);
} finally {
    if (pres != null) {
        pres.dispose();
    }
}
```

## **ШӘЩҶШёЫҢЩ… Ш№Щ„Ш§Щ…ШӘ ШҜШұШөШҜ ШҜШұ ШЁШұЪҶШіШЁвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ**

Ш§ЫҢЩҶ Ъ©ШҜ Ш¬Ш§ЩҲШ§Ш§ШіЪ©ШұЫҢЩҫШӘ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ ЪҶЪҜЩҲЩҶЩҮ Ш№Щ„Ш§Щ…ШӘ ШҜШұШөШҜ ШұШ§ ШЁШұШ§ЫҢ ЫҢЪ© ШЁШұЪҶШіШЁ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЫҢШҜ:
```javascript
// ЫҢЪ© ЩҶЩ…ЩҲЩҶЩҮ Ш§ШІ Ъ©Щ„Ш§Ші Presentation Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ
var pres = new aspose.slides.Presentation();
try {
    // Щ…ШұШ¬Ш№ ЫҢЪ© Ш§ШіЩ„Ш§ЫҢШҜ ШұШ§ Ш§ШІ Ш·ШұЫҢЩӮ Ш§ЫҢЩҶШҜЪ©Ші ШўЩҶ ШҜШұЫҢШ§ЩҒШӘ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    var slide = pres.getSlides().get_Item(0);
    // ЩҶЩ…ЩҲШҜШ§Шұ PercentsStackedColumn ШұШ§ ШЁШұ ШұЩҲЫҢ Ш§ШіЩ„Ш§ЫҢШҜ Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    var chart = slide.getShapes().addChart(aspose.slides.ChartType.PercentsStackedColumn, 20, 20, 500, 400);
    // NumberFormatLinkedToSource ШұШ§ ШЁЩҮ false ШӘЩҶШёЫҢЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ
    chart.getAxes().getVerticalAxis().setNumberFormatLinkedToSource(false);
    chart.getAxes().getVerticalAxis().setNumberFormat("0.00%");
    chart.getChartData().getSeries().clear();
    var defaultWorksheetIndex = 0;
    // ЩҲШұЪ©вҖҢШҙЫҢШӘ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ ШұШ§ ШҜШұЫҢШ§ЩҒШӘ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    var workbook = chart.getChartData().getChartDataWorkbook();
    // ШіШұЫҢ Ш¬ШҜЫҢШҜЫҢ Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    var series = chart.getChartData().getSeries().add(workbook.getCell(defaultWorksheetIndex, 0, 1, "Reds"), chart.getType());
    series.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 1, 1, 0.3));
    series.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 2, 1, 0.5));
    series.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 3, 1, 0.8));
    series.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 4, 1, 0.65));
    // ШұЩҶЪҜ ЩҫШұ Ъ©ШұШҜЩҶ ШіШұЫҢ ШұШ§ ШӘЩҶШёЫҢЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ
    series.getFormat().getFill().setFillType(java.newByte(aspose.slides.FillType.Solid));
    series.getFormat().getFill().getSolidFillColor().setColor(java.getStaticFieldValue("java.awt.Color", "RED"));
    // ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ LabelFormat ШұШ§ ШӘЩҶШёЫҢЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ
    series.getLabels().getDefaultDataLabelFormat().setShowValue(true);
    series.getLabels().getDefaultDataLabelFormat().setNumberFormatLinkedToSource(false);
    series.getLabels().getDefaultDataLabelFormat().setNumberFormat("0.0%");
    series.getLabels().getDefaultDataLabelFormat().getTextFormat().getPortionFormat().setFontHeight(10);
    series.getLabels().getDefaultDataLabelFormat().getTextFormat().getPortionFormat().getFillFormat().setFillType(java.newByte(aspose.slides.FillType.Solid));
    series.getLabels().getDefaultDataLabelFormat().getTextFormat().getPortionFormat().getFillFormat().getSolidFillColor().setColor(java.getStaticFieldValue("java.awt.Color", "WHITE"));
    series.getLabels().getDefaultDataLabelFormat().setShowValue(true);
    // ШіШұЫҢ Ш¬ШҜЫҢШҜЫҢ Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    var series2 = chart.getChartData().getSeries().add(workbook.getCell(defaultWorksheetIndex, 0, 2, "Blues"), chart.getType());
    series2.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 1, 2, 0.7));
    series2.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 2, 2, 0.5));
    series2.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 3, 2, 0.2));
    series2.getDataPoints().addDataPointForBarSeries(workbook.getCell(defaultWorksheetIndex, 4, 2, 0.35));
    // ЩҶЩҲШ№ ЩҫШұ Ъ©ШұШҜЩҶ ЩҲ ШұЩҶЪҜ ШұШ§ ШӘЩҶШёЫҢЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ
    series2.getFormat().getFill().setFillType(java.newByte(aspose.slides.FillType.Solid));
    series2.getFormat().getFill().getSolidFillColor().setColor(java.getStaticFieldValue("java.awt.Color", "BLUE"));
    series2.getLabels().getDefaultDataLabelFormat().setShowValue(true);
    series2.getLabels().getDefaultDataLabelFormat().setNumberFormatLinkedToSource(false);
    series2.getLabels().getDefaultDataLabelFormat().setNumberFormat("0.0%");
    series2.getLabels().getDefaultDataLabelFormat().getTextFormat().getPortionFormat().setFontHeight(10);
    series2.getLabels().getDefaultDataLabelFormat().getTextFormat().getPortionFormat().getFillFormat().setFillType(java.newByte(aspose.slides.FillType.Solid));
    series2.getLabels().getDefaultDataLabelFormat().getTextFormat().getPortionFormat().getFillFormat().getSolidFillColor().setColor(java.getStaticFieldValue("java.awt.Color", "WHITE"));
    // Ш§ШұШ§ШҰЩҮ ШұШ§ ШЁШұ ШұЩҲЫҢ ШҜЫҢШіЪ© Ш°Ш®ЫҢШұЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ
    pres.save("SetDataLabelsPercentageSign_out.pptx", aspose.slides.SaveFormat.Pptx);
} finally {
    if (pres != null) {
        pres.dispose();
    }
}
```

## **ШӘЩҶШёЫҢЩ… ЩҒШ§ШөЩ„ЩҮ ШЁШұЪҶШіШЁвҖҢЩҮШ§ Ш§ШІ Щ…ШӯЩҲШұ**

```javascript
// ЫҢЪ© ЩҶЩ…ЩҲЩҶЩҮ Ш§ШІ Ъ©Щ„Ш§Ші Presentation Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ
```

## **ШӘЩҶШёЫҢЩ… Щ…ЩҲЩӮШ№ЫҢШӘ ШЁШұЪҶШіШЁ**

ЩҮЩҶЪҜШ§Щ…ЫҢ Ъ©ЩҮ ЩҶЩ…ЩҲШҜШ§ШұЫҢ Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Ъ©ЩҮ ШЁЩҮ ЩҮЫҢЪҶ Щ…ШӯЩҲШұЫҢ ЩҲШ§ШЁШіШӘЩҮ ЩҶЫҢШіШӘШҢ Щ…Ш§ЩҶЩҶШҜ ЩҶЩ…ЩҲШҜШ§Шұ ЩҫШ§ЫҢШҢ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШЁШұЪҶШіШЁвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮЩ” ЩҶЩ…ЩҲШҜШ§Шұ ШЁШіЫҢШ§Шұ ЩҶШІШҜЫҢЪ© ШЁЩҮ ШӯШ§ШҙЫҢЩҮЩ” ШўЩҶ ШҙЩҲЩҶШҜ. ШҜШұ ЪҶЩҶЫҢЩҶ Щ…ЩҲШ§ШұШҜЫҢ ШЁШ§ЫҢШҜ Щ…ЩҲЩӮШ№ЫҢШӘ ШЁШұЪҶШіШЁ ШҜШ§ШҜЩҮ ШұШ§ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЫҢШҜ ШӘШ§ Ш®Ш·ЩҲШ· ШұШ§ЩҮЩҶЩ…Ш§ ШЁЩҮвҖҢЩҲШ¶ЩҲШӯ ЩҶЩ…Ш§ЫҢШҙ ШҜШ§ШҜЩҮ ШҙЩҲЩҶШҜ.
Ш§ЫҢЩҶ Ъ©ШҜ Ш¬Ш§ЩҲШ§Ш§ШіЪ©ШұЫҢЩҫШӘ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ ЪҶЪҜЩҲЩҶЩҮ Щ…ЩҲЩӮШ№ЫҢШӘ ШЁШұЪҶШіШЁ ШұШ§ ШҜШұ ЫҢЪ© ЩҶЩ…ЩҲШҜШ§Шұ ЩҫШ§ЫҢ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЫҢШҜ:
```javascript
var pres = new aspose.slides.Presentation();
try {
    var chart = pres.getSlides().get_Item(0).getShapes().addChart(aspose.slides.ChartType.Pie, 50, 50, 200, 200);
    var series = chart.getChartData().getSeries();
    var label = series.get_Item(0).getLabels().get_Item(0);
    label.getDataLabelFormat().setShowValue(true);
    label.getDataLabelFormat().setPosition(aspose.slides.LegendDataLabelPosition.OutsideEnd);
    label.setX(0.71);
    label.setY(0.04);
    pres.save("pres.pptx", aspose.slides.SaveFormat.Pptx);
} finally {
    if (pres != null) {
        pres.dispose();
    }
}
```

## **ШіШӨШ§Щ„Ш§ШӘ Щ…ШӘШҜШ§ЩҲЩ„**

**ЪҶЪҜЩҲЩҶЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩ… Ш§ШІ ЩҮЩ…вҖҢЩҫЩҲШҙШ§ЩҶЫҢ ШЁШұЪҶШіШЁвҖҢЩҮШ§ЫҢ ШҜШ§ШҜЩҮ ШҜШұ ЩҶЩ…ЩҲШҜШ§ШұЩҮШ§ЫҢ ЩҫШұШ¬Щ…Ш№ЫҢШӘ Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ъ©ЩҶЩ…Шҹ**

Ш§ШІ Ш¬Ш§ЫҢЪҜШ°Ш§ШұЫҢ Ш®ЩҲШҜЪ©Ш§Шұ ШЁШұЪҶШіШЁвҖҢЩҮШ§ШҢ Ш®Ш·ЩҲШ· ШұШ§ЩҮЩҶЩ…Ш§ ЩҲ Ъ©Ш§ЩҮШҙ Ш§ЩҶШҜШ§ШІЩҮ ЩӮЩ„Щ… Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜШӣ ШҜШұ ШөЩҲШұШӘ Щ„ШІЩҲЩ… ШЁШұШ®ЫҢ ЩҒЫҢЩ„ШҜЩҮШ§ (Щ…Ш«Щ„Ш§ЩӢ ШҜШіШӘЩҮвҖҢШЁЩҶШҜЫҢ) ШұШ§ Щ…Ш®ЩҒЫҢ Ъ©ЩҶЫҢШҜ ЫҢШ§ ЩҒЩӮШ· ШЁШұШ§ЫҢ ЩҶЩӮШ§Ш· Ш§ЩҶШӘЩҮШ§ЫҢЫҢ/Ъ©Щ„ЫҢШҜЫҢ ШЁШұЪҶШіШЁвҖҢЩҮШ§ ШұШ§ ЩҶЩ…Ш§ЫҢШҙ ШҜЩҮЫҢШҜ.

**ЪҶЪҜЩҲЩҶЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩ… ШЁШұЪҶШіШЁвҖҢЩҮШ§ ШұШ§ ЩҒЩӮШ· ШЁШұШ§ЫҢ Щ…ЩӮШ§ШҜЫҢШұ ШөЩҒШұШҢ Щ…ЩҶЩҒЫҢ ЫҢШ§ Ш®Ш§Щ„ЫҢ ШәЫҢШұЩҒШ№Ш§Щ„ Ъ©ЩҶЩ…Шҹ**

ЩҶЩӮШ§Ш· ШҜШ§ШҜЩҮ ШұШ§ ЩҫЫҢШҙ Ш§ШІ ЩҒШ№Ш§Щ„вҖҢШіШ§ШІЫҢ ШЁШұЪҶШіШЁвҖҢЩҮШ§ ЩҒЫҢЩ„ШӘШұ Ъ©ЩҶЫҢШҜ ЩҲ ЩҶЩ…Ш§ЫҢШҙ Щ…ЩӮШ§ШҜЫҢШұ ШөЩҒШұШҢ Щ…ЩӮШ§ШҜЫҢШұ Щ…ЩҶЩҒЫҢ ЫҢШ§ Щ…ЩӮШ§ШҜЫҢШұ ЪҜЩ…ШҙШҜЩҮ ШұШ§ ШЁШұ Ш§ШіШ§Ші ЫҢЪ© ЩӮШ§ЩҶЩҲЩҶ ШӘШ№ШұЫҢЩҒвҖҢШҙШҜЩҮ ШәЫҢШұЩҒШ№Ш§Щ„ Ъ©ЩҶЫҢШҜ.

**ЪҶЪҜЩҲЩҶЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩ… Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ШӯШ§ШөЩ„ Ъ©ЩҶЩ… Ъ©ЩҮ ШіШЁЪ© ШЁШұЪҶШіШЁ ЩҮЩҶЪҜШ§Щ… Ш®ШұЩҲШ¬ЫҢ ШЁЩҮ PDF/ШӘШөШ§ЩҲЫҢШұ Ш«Ш§ШЁШӘ ШЁШ§ШҙШҜШҹ**

ЩӮЩ„Щ…вҖҢЩҮШ§ (ЩҶШ§Щ… Ш®Ш§ЩҶЩҲШ§ШҜЪҜЫҢШҢ Ш§ЩҶШҜШ§ШІЩҮ) ШұШ§ ШЁЩҮвҖҢШөЩҲШұШӘ ШөШұЫҢШӯ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЫҢШҜ ЩҲ Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ШӯШ§ШөЩ„ Ъ©ЩҶЫҢШҜ Ъ©ЩҮ ЩӮЩ„Щ… ШЁШұ ШұЩҲЫҢ ШіЩ…ШӘ ШұЩҶШҜШұ Щ…ЩҲШ¬ЩҲШҜ Ш§ШіШӘ ШӘШ§ Ш§ШІ Ш§ШіШӘЩҒШ§ШҜЩҮЩ” ЩӮЩ„Щ… ЩҫЫҢШҙвҖҢЩҒШұШ¶ Ш¬Щ„ЩҲЪҜЫҢШұЫҢ ШҙЩҲШҜ.
